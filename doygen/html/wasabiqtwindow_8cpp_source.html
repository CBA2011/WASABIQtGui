00041 #define SPACES " \t\r\n"
00042 using namespace std;
00043 inline string trim_right(const string & s, const string & t = SPACES) {
00044     string d(s);
00045     string::size_type i(d.find_last_not_of(t));
00046     if (i == string::npos)
00047         return "";
00048     else
00049         return d.erase(d.find_last_not_of(t) + 1);
00050 } // end of trim_right
00051 
00052 inline string trim_left(const string & s, const string & t = SPACES) {
00053     string d(s);
00054     return d.erase(0, s.find_first_not_of(t));
00055 } // end of trim_left
00056 
00057 inline string trim(const string & s, const string & t = SPACES) {
00058     string d(s);
00059     return trim_left(trim_right(d, t), t);
00060 } // end of trim

00062 void myReplace(std::string& str, const std::string& oldStr, const std::string& newStr)
00063 {
00064     size_t pos = 0;
00065     while((pos = str.find(oldStr, pos)) != std::string::npos)
00066     {
00067         str.replace(pos, oldStr.length(), newStr);
00068         pos += newStr.length();
00069     }
00070 }

00733 bool WASABIQtWindow::parseMessage(QString message) {
00734     std::cout << "WASABIQtWindow::parseMessage: message = '" << message.toStdString() << "'" << std::endl;
00735     QStringList str_list = message.split("&");
00736     if (str_list.size() < 3) {
00737         std::cerr << "WASABIQtWindow::parseMessage: too few tokens in message!" << std::endl;
00738         return false;
00739     }
00740     QString senderID = str_list.at(0); // will be ignored so far, but must be supplied
00741     QString command = str_list.at(1);
00742     QString targetID = str_list.at(2); // or <name> for ADD or <typeOfInfo> for REQUEST
00743     QString param1;
00744     QString param2;
00745     std::cout << "WASABIQtWindow::parseMessage: now checking '" << senderID.toStdString()
00746               << "' with command '" << command.toStdString()
00747               << "' with target '"
00748               << targetID.toStdString() << "'." << std::endl;
00749     double lifetime = -1;
00750     int eaID;
00751     int newLocalID = 0;
00752     int impulse; // P, A, D, X, Y, Z;
00753     cogaEmotionalAttendee* ea = wasabi->getEAfromID(currentEA);
00754     if (!ea) {
00755         std::cout << "WASABIQtWindow::parseMessage: no emotionalAttendee with ID '" << currentEA << "' found!" << std::endl;
00756         return false;
00757     }
00758     bool ok = false;
00759     switch (returnIndex(command.toStdString(), "ADD TRIGGER IMPULSE")) {
00760     //<String senderID>|ADD|<String name==targetID>|<String globalID (optional)>
00761     //e.g. 'Robovie|ADD|Robovie|Robovie23' or 'Robovie1|ADD|Dylan F. Glas|120345_1' or simply 'Robovie|ADD|Chris'
00762     case 1:
00763         if (str_list.size() == 3) {
00764             newLocalID = addEmotionalAttendee(targetID, "undef"); // targetID is the 'real name' here, e.g. 'Dylan F. Glas'
00765         }
00766         else {
00767             param1 = str_list.at(3); // <String globalID>
00768             newLocalID = addEmotionalAttendee(targetID, param1); // param1 is the globalID here, e.g. '120345_1'
00769         }
00770         if (newLocalID == 0) {
00771             return false;
00772         }
00773 
00774         break;
00775     case 2: //TRIGGER <String targetID> <String affectiveStateName> <double lifetime>
00776         if (str_list.size()==3) {
00777             std::cout << "WASABIQtWindow::parseMessage: not enough tokens for '"
00778                       << command.toStdString() << "'!" << std::endl;
00779             return false;
00780         }
00781         param1 = str_list.at(3); // <String affectiveStateName>
00782         if (str_list.size() == 4) {
00783             std::cout << "WASABIQtWindow::parseMessage: no lifetime for '"
00784                       << command.toStdString() << "', using standard lifetime." << std::endl;
00785             param2 = "-1";
00786         }
00787         else {
00788             param2 = str_list.at(4); // <double lifetime>
00789         }
00790         lifetime = param2.toDouble(&ok); //atof((const char*)param2.mb_str());
00791         if (!ok || (lifetime < 0 && lifetime != -1) || lifetime > 100) {
00792             std::cout
00793                     << "WASABIQtWindow::parseMessage: second parameter (double lifetime) of '"
00794                     << command.toStdString()
00795                     << "' is not in range ]0, 100] and not -1 (or no double at all)!"
00796                     << std::endl;
00797             return false;
00798         }
00799 
00800         //TODO CHECK THE ABOVE!!!
00801         eaID = targetID.toInt(&ok);//atoi((const char*)targetID.mb_str());
00802         std::cout << "WASABIQtWindow::parseMessage: eaID = " << eaID << std::endl;
00803         if (ok && eaID > 0) {
00804             ea = wasabi->getEAfromID(eaID);
00805         }
00806         if (!ea || !ea->EmoConPerson->triggerAS(param1.toStdString(), lifetime)) {
00807             std::cout << "WASABIQtWindow::parseMessage: Couldn't TRIGGER '" << param1.toStdString() << "'!" << std::endl;
00808             return false;
00809         }
00810         break;
00811     case 3: //IMPULSE <String targetID> <int impulse>
00812         if (str_list.size()==3) {
00813             std::cout << "WASABIQtWindow::parseMessage: not enough tokens for '"
00814                       << command.toStdString() << "'!" << std::endl;
00815             return false;
00816         }
00817         param1 = str_list.at(3);
00818         impulse = param1.toInt(&ok); //atoi((const char*)param1.mb_str());
00819         if (!ok || impulse < -100 || impulse > 100 || impulse == 0) {
00820             std::cout
00821                     << "WASABIQtWindow::parseMessage: second parameter (int impulse) of '"
00822                     << command.toStdString()
00823                     << "' is not in range [-100, 100] or 0 (or no integer at all)!"
00824                     << std::endl;
00825             return false;
00826         }
00827         eaID = targetID.toInt(&ok);//atoi((const char*)targetID.mb_str());
00828         std::cout << "WASABIQtWindow::parseMessage: eaID = " << eaID << std::endl;
00829         return wasabi->emotionalImpulse(impulse, eaID);
00830         break;
00831     default:
00832         std::cout << "WASABIQtWindow::parseMessage: unknown command '" << command.toStdString()
00833                   << "'!" << std::endl;
